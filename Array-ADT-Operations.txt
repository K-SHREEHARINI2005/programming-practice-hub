void create_arr(int *a,int n)
{
   int i;

      scanf("%d",&a[i]);
}
void insert_arr(int *a,int *n)
{
   int pos,i,b;
   printf("enter the element:");
   scanf("%d",&b);
   printf("enter the position:");
   scanf("%d",&pos);
   if(pos<1||pos>*n+1)
   {
      printf("invalid position");
      return;
   }
   for(i=*n;i>=pos;i--)
      a[i]=a[i-1];
   a[pos-1]=b;
   (*n)++;
   printf("element after inserted:");
   for(i=0;i<(*n);i++)
      printf("%d\n",a[i]);
}
void delete_arr(int *a,int *n)
{
   int c,i;
   printf("enter the position:");
   scanf("%d",&c);
   if(c<1||c>*n)
   {
      printf("there is no element in the position");
      return;
   }
   for(i=c-1;i<*n-1;i++)
      a[i]=a[i+1];
   (*n)--;
   printf("elements after deleted:");
   for(i=0;i<(*n);i++)
      printf("%d\n",a[i]);
}
