#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <unistd.h>
#include <arpa/inet.h>
#include <time.h>

#define PORT 6799
#define MAX_PKT 10

int main() {
    int sockfd, newsockfd;
    struct sockaddr_in server_addr, client_addr;
    socklen_t addr_len = sizeof(client_addr);
    char buffer[1024];
    int expected_seq = 0;
    int simulate_loss = 0;
    int loss_prob = 0;

    srand(time(NULL));

    // User menu for packet loss
    printf("Do you want to simulate packet loss? (1=Yes, 0=No): ");
    scanf("%d", &simulate_loss);
    if (simulate_loss) {
        printf("Enter packet loss probability (0-100): ");
        scanf("%d", &loss_prob);
    }

    sockfd = socket(AF_INET, SOCK_STREAM, 0);
    if (sockfd < 0) { perror("Socket"); exit(1); }

    server_addr.sin_family = AF_INET;
    server_addr.sin_addr.s_addr = INADDR_ANY;
    server_addr.sin_port = htons(PORT);

    if (bind(sockfd, (struct sockaddr*)&server_addr, sizeof(server_addr)) < 0) {
        perror("Bind"); exit(1); }

    listen(sockfd, 5);
    printf("Server listening on port %d...\n", PORT);

    newsockfd = accept(sockfd, (struct sockaddr*)&client_addr, &addr_len);
    if (newsockfd < 0) { perror("Accept"); exit(1); }

    while (1) {
        int n = recv(newsockfd, buffer, sizeof(buffer), 0);
        if (n <= 0) break;

        int seq;
        sscanf(buffer, "%d", &seq);

        // Simulate packet loss if enabled
        if (simulate_loss) {
            int r = rand() % 100;
            if (r < loss_prob) {
                printf("Packet %d lost!\n", seq);
                continue; // do not send ACK
            }
        }

        printf("Received packet %d\n", seq);

        if (seq == expected_seq) {
            printf("Packet %d accepted\n", seq);
            expected_seq++;
        } else {
            printf("Packet %d discarded, expected %d\n", seq, expected_seq);
        }

        // Send ACK for last correctly received packet
        sprintf(buffer, "%d", expected_seq - 1);
        send(newsockfd, buffer, strlen(buffer) + 1, 0);
    }

    close(newsockfd);
    close(sockfd);
    return 0;
}
