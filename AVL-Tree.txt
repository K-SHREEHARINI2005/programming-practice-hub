#include "avl.h"
int max(int a, int b) {
    return (a > b) ? a : b;
}

int height(avl t) {
    if (t == NULL)
        return 0;
    return t->height;
}

avl sll(avl k2) {
    avl k1 = k2->left;
    k2->left = k1->right;
    k1->right = k2;
    k2->height = max(height(k2->left), height(k2->right)) + 1;
    k1->height = max(height(k1->left), height(k1->right)) + 1;
    return k1;
}

avl srr(avl k1) {
    avl k2 = k1->right;
    k1->right = k2->left;
    k2->left = k1;
    k1->height = max(height(k1->left), height(k1->right)) + 1;
    k2->height = max(height(k2->left), height(k2->right)) + 1;
    return k2;
}

avl dlr(avl k3) {
    k3->left = srr(k3->left);
    return sll(k3);
}

avl drl(avl k1) {
    k1->right = sll(k1->right);
    return srr(k1);
}

avl insert(int x, avl t) {
    if (t == NULL) {
        t = (avl)malloc(sizeof(struct node));
        t->data = x;
        t->left = NULL;
        t->right = NULL;
        t->height = 1;
    } else if (x < t->data) {
        t->left = insert(x, t->left);
        if (height(t->left) - height(t->right) == 2) {
            if (x < t->left->data)
                t = sll(t);
            else
                t = dlr(t);
        }
    } else if (x > t->data) {
        t->right = insert(x, t->right);
        if (height(t->right) - height(t->left) == 2) {
            if (x > t->right->data)
                t = srr(t);
            else
                t = drl(t);
        }
    }

    t->height = max(height(t->left), height(t->right)) + 1;
    return t;
}

avl findmin(avl t) {
    if (t == NULL)
        return NULL;
    while (t->left != NULL)
        t = t->left;
    return t;
}

avl findmax(avl t) {
    if (t == NULL)
        return NULL;
    while (t->right != NULL)
        t = t->right;
    return t;
}

avl delete(int x, avl t) {
    avl temp;

    if (t == NULL) {
        printf("Element not found.\n");
        return NULL;
    } else if (x < t->data) {
        t->left = delete(x, t->left);
        if (height(t->right) - height(t->left) == 2) {
            if (height(t->right->left) > height(t->right->right))
                t = drl(t);
            else
                t = srr(t);
        }
    } else if (x > t->data) {
        t->right = delete(x, t->right);
        if (height(t->left) - height(t->right) == 2) {
            if (height(t->left->right) > height(t->left->left))
                t = dlr(t);
            else
                t = sll(t);
        }
    } else {
        if (t->left != NULL && t->right != NULL) {
            temp = findmin(t->right);
            t->data = temp->data;
            t->right = delete(t->data, t->right);
        } else {
            temp = t;
            if (t->left == NULL)
                t = t->right;
            else if (t->right == NULL)
                t = t->left;
            free(temp);
        }
    }

    if (t != NULL)
        t->height = max(height(t->left), height(t->right)) + 1;

    return t;
}

avl search(int x, avl t) {
    if (t == NULL) {
        return NULL;
    } else if (x < t->data) {
        return search(x, t->left);
    } else if (x > t->data) {
        return search(x, t->right);
    } else {
        return t;
    }
}

    if (t != NULL) {
        inorder(t->left);

        preorder(t->left);

void postorder(avl t) {
    if (t != NULL) {
        postorder(t->left);
        postorder(t->right);
        printf("%d ", t->data);
    }
}

    printf("Inorder Traversal: ");
