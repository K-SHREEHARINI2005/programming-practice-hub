#include <stdio.h>
#include <stdlib.h>
#include <stdbool.h>
#define MAX 100
 typedef struct {
 char items[MAX];
 int top;
} Stack;

#include"stk.h"
bool isEmpty(Stack *s) {
    return s->top == -1;
}
void initializeStack(Stack *s) {
       s->top = -1;
}
bool isFull(Stack *s) {
    return s->top == MAX - 1;
}

void push(Stack *s, char value) {
    if(isFull(s))
    {
        printf("Stack Overflow\n");
        exit(1);
    }
    else
    {
        s->items[(s->top)++]=value;
    }
}

char pop(Stack *s) {
    if (isEmpty(s)) {
        printf("Stack Underflow\n");
        exit(1);
    } else {
        char value = s->items[(s->top)--];
        printf("Popped from the stack.\n", &value);
        return value;
    }
}

void display(Stack *s) {
   int i;
    if (isEmpty(s)) {
        printf("Stack is empty\n");
    } else {
        printf("Stack contents: ");
        for (i = 0; i <= s->top; i++) {
            printf("%c ", s->items[i]);
        }
        printf("\n");
    }
}

bool areParenthesesBalanced(Stack *s, char expr[]) {
    int i;
    for (i = 0; expr[i] != '\0'; i++) {
        char current = expr[i];

        if (current == '(' || current == '{' || current == '[') {
            push(s, current);
        } else if (current == ')' || current == '}' || current == ']') {
            if (isEmpty(s)) {
                return false;
            }

            char popped = pop(s);
            if ((current == ')' && popped != '(') ||
                (current == '}' && popped != '{') ||
                (current == ']' && popped != '[')) {
                return false;
            }
        }
    }

    return isEmpty(s);
}
