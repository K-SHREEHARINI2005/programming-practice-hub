#include <stdio.h>
#include <string.h>
int main() {
    FILE *a, *b;
    char input[100], output[200], flag[10];
    int i, j = 0, count = 0;
    printf("Enter bit flag : ");
    scanf("%s", flag);
    a = fopen("bist.c", "r");
    if (a == NULL) {
        printf("Cannot open input file\n");
        return 1;
    }
    fgets(input, sizeof(input), a);
    fclose(a);
    strcpy(output, flag);
    j = strlen(flag);
    for (i = 0; i < strlen(input); i++) {
        output[j++] = input[i];

        if (input[i] == '1') {
            count++;
            if (count == 5) {
                output[j++] = '0';
                count = 0;
            }
        } else {
            count = 0;
        }
    }
    strcat(output, flag);
    b = fopen("bisto.c", "w");
    if (b == NULL) {
        printf("Cannot open output file\n");
        return 1;
    }
    fputs(output, b);
    fclose(b);
    return 0;
}
