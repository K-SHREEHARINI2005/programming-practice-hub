#include <stdio.h>
#include <limits.h>
#include <stdbool.h>

#define INF INT_MAX

void dijkstra(int nodes, int graph[][nodes], int src, int dist[]) {
    bool visited[nodes];
    int i, count, u, v, min;

    for (i = 0; i < nodes; i++) {
        dist[i] = INF;
        visited[i] = false;
    }
    dist[src] = 0;

    for (count = 0; count < nodes - 1; count++) {
        min = INF;
        for (v = 0; v < nodes; v++) {
            if (!visited[v] && dist[v] <= min) {
                min = dist[v];
                u = v;
            }
        }
        visited[u] = true;
        for (v = 0; v < nodes; v++) {
            if (!visited[v] && graph[u][v] && dist[u] != INF &&
                dist[u] + graph[u][v] < dist[v]) {
                dist[v] = dist[u] + graph[u][v];
            }
        }
    }
}

void routing(int nodes, int graph[][nodes]) {
    int dist[nodes];
    int src, dest, nextHop, i;

    printf("\n--- Routing Tables ---\n");
    for (src = 0; src < nodes; src++) {
        dijkstra(nodes, graph, src, dist);
        printf("\nRouter %d:\n", src);
        printf("Destination\tCost\tNext Hop\n");
        for (dest = 0; dest < nodes; dest++) {
            if (dest != src) {
                nextHop = dest;
                for (i = 0; i < nodes; i++) {
                    if (graph[src][i] != 0 && dist[dest] == dist[i] + graph[i][dest]) {
                        nextHop = i;
                    }
                }
                printf("%d\t\t%d\t%d\n", dest, dist[dest], nextHop);
            }
        }
    }
}

int main() {
    int nodes;
    printf("Enter the number of routers: ");
    scanf("%d", &nodes);
    int graph[nodes][nodes];
    int i, j;

    printf("Enter the adjacency matrix (0 for no direct link):\n");
    for (i = 0; i < nodes; i++) {
        for (j = 0; j < nodes; j++) {
            printf("Link cost from %d to %d: ", i, j);
            scanf("%d", &graph[i][j]);
            if (graph[i][j] == 0 && i != j) {
                graph[i][j] = INF;
            }
        }
    }
    routing(nodes, graph);
    return 0;
}
