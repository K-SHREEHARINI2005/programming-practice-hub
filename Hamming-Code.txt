#include <stdio.h>
#include <string.h>

int powerOfTwo(int m) {
    int result = 1;
    int i;
    for ( i = 0; i < m; i++) result *= 2;
    return result;
}

int calParity(int m) {
    int r = 0;
    while ((1 << r) < (m + r + 1)) {
        r++;
    }
    return r;
}

int main() {
    FILE *fp;
    char data[100], code[200];
    int m, r, totalbits, i, j;

    printf("Enter binary data: ");
    scanf("%s", data);

    m = strlen(data);
    r = calParity(m);
    totalbits = m + r;

    j = m - 1;
    for (i = 1; i <= totalbits; i++) {
        if ((i & (i - 1)) == 0) {
            code[i] = '0';
        } else {
            code[i] = data[j--];
        }
    }
    for (i = 0; i < r; i++) {
        int pos = powerOfTwo(i);
        int count = 0;
        for (j = 1; j <= totalbits; j++) {
            if (j & pos) {
                if (code[j] == '1') count++;
            }
        }
        if (count % 2 != 0) {
            code[pos] = '1';
        }
    }
    code[totalbits + 1] = '\0';
    fp = fopen("ham.c", "w");
    if (!fp) {
        printf("Cannot open file\n");
        return 1;
    }
    for (i = 1; i <= totalbits; i++) {
        fputc(code[i], fp);
    }
    fclose(fp);
    printf("Hamming code written: ");
    return 0;
}

#include <stdio.h>
#include <string.h>
#include <math.h>

int powoftwo(int m) {
    int result = 1;
    int i;
    for ( i = 0; i < m; i++) {
        result = result * 2;
    }
    return result;
}

int main() {
    FILE *fp;
    char code[200];
    int totalbits, r, i, j, error = 0;

    fp = fopen("ham.c", "r");
    if (!fp) {
        printf("cannot open\n");
        return 1;
    }
    fgets(code, sizeof(code), fp);
    fclose(fp);
    totalbits = strlen(code) - 1;
    r = 0;
    while (powoftwo(r) <= totalbits) r++;

    for (i = 0; i < r; i++) {
        int pos = powoftwo(i);
        int count = 0;

        for (j = 1; j <= totalbits; j++) {
            if (j & pos) {
                if (code[j - 1] == '1') count++;
            }
        }
        if (count % 2 != 0) {
            error += pos;
        }
    }

    if (error == 0) {
        printf("No error\n");
    } else {
        printf("no error");
    }

    return 0;
}
