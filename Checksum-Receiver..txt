
#include <stdio.h>
#include <string.h>
void binaryAdd4Bit(char a[], char b[], char result[]) {
    int carry = 0, sum,i;
    for ( i = 3; i >= 0; i--) {
        int bitA = a[i] - '0';
        int bitB = b[i] - '0';
        sum = bitA + bitB + carry;
        result[i] = (sum % 2) + '0';
        carry = sum / 2;
    }
    result[4] = '\0';
    if (carry == 1) {
        char one[5] = "0001";
        char temp[5];
        strcpy(temp, result);
        binaryAdd4Bit(temp, one, result);
    }
}
void onesComplement4Bit(char data[]) {
 int i;
    for ( i = 0; i < 4; i++) {
        data[i] = (data[i] == '0') ? '1' : '0';
    }
    data[4] = '\0';
}

    FILE *fp;
    char received[1000], chunks[100][5], sum[5] = "0000";

    fscanf(fp, "%s", received);

    int len = strlen(received);
    int totalChunks = len / 4;
    int i;
    for ( i = 0; i < totalChunks; i++) {
        strncpy(chunks[i], received + i * 4, 4);
        chunks[i][4] = '\0';
    }
    for ( i = 0; i < totalChunks; i++) {
        char tempSum[5];
        binaryAdd4Bit(sum, chunks[i], tempSum);
        strcpy(sum, tempSum);
    }
    onesComplement4Bit(sum);
    printf("Final 1's Complement of Sum: %s\n", sum);
