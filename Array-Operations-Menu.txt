#include<stdio.h>
int main()
{
   int a[10],choice,n,temp,i,j,median,x;

   printf("1.ascending order\n");

     }

   {
      case 1:
         printf("\n ascending order is \n");
         for(i=0;i<n;i++)
         {
            for(j=i+1;j<n;j++)
            {
               if(a[i]>a[j])
               {
                  temp =a[i];
                  a[i]=a[j];

               }
            }
         }

            printf("%d",a[i]);
         }
         break;
      case 2:
          printf("\n descending order is \n");
          for(i=0;i<n;i++)
          {
             for(j=i+1;j<n;j++)
             {
               if(a[i]<a[j])
              {
                 temp =a[i];
                 a[i]=a[j];
                 a[j]=temp;
              }
             }
          }
          for(i=0;i<n;i++)
          {
             printf("%d",a[i]);
          }
          break;
      case 3:
           printf("\n smallest number is \n");
            for(i=0;i<n;i++)
            {
               for(j=i+1;j<n;j++)
               {
                     if(a[i]>a[j])
                     {
                       temp =a[i];
                       a[i]=a[j];
                       a[j]=temp;
                     }
                 }
             }
                   printf("%d",a[0]);

      case 4:
               printf("\n largest number is \n");

                  for(j=i+1;j<n;j++)
                  {

                                   temp =a[i];

                            }

                   printf("%d",a[0]);

                   printf("median=%d",a[x]);

                   printf("median=%d",(a[x]+a[x-i])/2.0);
                for(i=0;i<n;i++)
                {
                   a[i]=a[i];
                }
                break;
             }
         }
         printf("%f",median);
         break;
      default:
            printf("invalid");
   }}while(choice!=6);

   }
